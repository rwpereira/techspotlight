function getSasUrl(uuid: string) {
  // Sanitiza UUID antes do path
  const cleanId = uuid
    .replace(/-/g, '')

  const blobPath =
    `${container}/${cleanId}`

  return generateSasToken(blobPath)
}
#11933 — Migração de upload: memoryStorage → diskStorage
Upload de arquivo de 150 MB causou crash por OOM — o Multer carregava o arquivo inteiro na RAM do servidor NestJS. Endpoints migrados para diskStorage com diretório temporário. Arquivos deletados via try/finally para garantir cleanup mesmo em caso de erro.
ANTES — memoryStorage

150MB → heap overflow
// servidor crash 💥
DEPOIS — diskStorage
arquivo → disco tmpdir()
stream → Azure Blob
// finally { unlink(tmp) }

async uploadToBlob(file) {
  try {
    await streamToBlob(file.path)
  } finally {
    // Sempre limpa o arquivo temp
    unlinkSync(file.path)
  }
}
